// Parse JSON by default
app.use(express.json());

// Error Handling

app.all('/*', (req, res) => {
    res.status(404).send({ msg: 'Error 404: Route not found' });
});
 
app.use((err, req, res, next) => {
    console.log(err);
    res.sendStatus(500)
         .send ({msg : 'Error 500: Internal Server Error'});
});
